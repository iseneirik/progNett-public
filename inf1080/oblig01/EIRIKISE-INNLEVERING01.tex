\subsection*{Oppgave 1.9}

\begin{description}
\item[(a)] $A\setminus B = \{5,7,9\}$ 
\item[(b)] $B\setminus A = \{0,2,4\}$
\item[(c)] $(A\cup B) = \{5,7,9\}$
\item[(d)] $C\setminus (A\cup B) = \{6,8\}$
\item[(e)] $(A\setminus B)\setminus C = \emptyset$
\item[(f)] $(B\cup C)\setminus A = \{0,2,4,6,8\}$
\end{description}
\end{multicols}

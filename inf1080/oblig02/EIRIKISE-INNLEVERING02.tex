\subsection*{Oppgave 3.8 (a)}
\emph{H:} \textbf{``Jeg spiser honning''}\tab \emph{M:} \textbf{``Jeg drikker melk''}\tab \emph{B:} \textbf{``Jeg spiser brød''}
\begin{description}
\item[(1)] $(H \land M) \rightarrow \neg B$
\item[(2)] $(\neg B \land \neg H) \land M $
\end{description}


\subsection*{Oppgave 3.8 (b)}
\begin{description}
\item[(1)] \textbf{Hvis} jeg drikker melk hver gang jeg spiser honning, \textbf{så} spiser jeg brød 
\item[(2)] \textbf{Hvis} jeg spiser honning, \textbf{så} er det slik at jeg spiser brød \textbf{hvis} jeg drikker melk
\end{description}
